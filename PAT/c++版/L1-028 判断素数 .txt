#include<iostream>
#include<cmath>
using namespace std;
bool IsPrime(int n) {
	if (n < 2)
		return false;
	for (int i = 2; i <= sqrt(n); i++) {
		if (n%i == 0) {
			return false;
		}
	}
	return true;
}
int main() {
	int n;
	cin >> n;
	int num;
	for (int i = 0; i < n; i++) {
		cin >> num;
		if (IsPrime(num)) {
			cout << "Yes" << endl;
		}
		else
			cout << "No" << endl;
	}
	system("pause");
	return 0;
}
